    **function renderTableBody(rows):**
    - Per POL-03: build complete HTML string in memory, single insertAdjacentHTML
    ```javascript
    function renderTableBody(rows) {
      const cols = state.visibleColumns || new Set(state.columns);
      let html = '';
      for (const row of rows) {
        html += '<tr>';
        for (const col of state.columns) {
          if (!cols.has(col)) continue;
          const value = row.values[col] || '';
          const band = row.bands[col] || '';
          const flag = row.flags[col] || '';
          const { bg, text } = getCellColors(band, flag, state.editionDate);
          const displayValue = state.searchQuery ? highlightMatch(value, state.searchQuery) : escapeHtml(value);
          const bgStyle = bg ? `background:${bg};color:${text}` : '';
          html += `<td${bgStyle ? ` style="${bgStyle}"` : ''}>${displayValue}</td>`;
        }
        html += '</tr>';
      }
      el.tableBody.replaceChildren();
      el.tableBody.insertAdjacentHTML('beforeend', html);
    }
    ```
